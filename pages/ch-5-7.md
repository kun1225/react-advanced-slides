# 如何使用 React Compiler？

如果你使用的是 Next.js 16，啟用 React Compiler 的流程相當簡單。

首先，安裝所需的套件：

```bash
npm install -D babel-plugin-react-compiler
```

````md magic-move
  reactCompiler: true, // 👈 啟用 React Compiler
````

# React Compiler 的底層實踐原理

<v-click>

````md magic-move
```jsx
function Username({ name }) {
  console.log('render Username component');

  return (
    <div>
      <p>{name}'s todo</p>
    </div>
  );
}
```

```jsx
function Username({ name }) {
  console.log('render Username component');

  return __jsx('div', null, __jsx('p', null, name, "'s todo"));
  // ☝️ 將 jsx 轉換成 JS 物件
}
```

```jsx {all|7|4-5,9-10|11|12|13|15-16}
function Username(t0) {
  const $ = _c(2);

  // 從 Props 取出 name
  const { name } = t0;
  console.log('render Username component');
  let t1; // 👈 用來儲存 __jsx 結果的變數

  // 👇 當 props - name 有變更時
  if ($[0] !== name) {
    t1 = __jsx('div', null, __jsx('p', null, name, "'s todo")); // 👈 計算新的 t1
    $[0] = name; // 👈 將 name 記錄在 $[0]
    $[1] = t1; // 👈 將 t1 記錄在 $[1]
  } else {
    // 👈 當 props - name 沒有變更時
    t1 = $[1]; // 👈 直接取得記錄過的 $[1]
  }

  return t1;
}
```
````

# React Compiler 的底層實踐原理

<v-click>

````md magic-move
```tsx
function Counter() {
  const [count, setCount] = useState(1);

  return (
    <div>
      <p>{count}</p>
    </div>
  );
}
```

```tsx
function Counter() {
  const $ = _c(2);

  const [count] = useState(1);
  let t0;

  // 👇 當 state - count 有變更時才會計算新的值
  if ($[0] !== count) {
    t0 = __jsx('div', null, __jsx('p', null, count));
    $[0] = count;
    $[1] = t0;
  } else {
    t0 = $[1];
  }

  return t0;
}
```
````
