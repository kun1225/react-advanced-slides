  const [activeSection, setActiveSection] = useState('');
  const [isBarVisible, setIsBarVisible] = useState(false);
  const [scrollProgress, setScrollProgress] = useState(0);

  useEffect(() => {
    // 監聽滾動事件與邏輯
  }, []);

  <div className="profile-container" ref={containerRef}>
    <DynamicTitle activeSection={activeSection} />

    {children}

    <FloatingBar
      isVisible={isBarVisible}

# 為什麼 children 不會 re-render？

<img src="/ch-5/5-2/why.png" class="absolute inset-0" />

<!--
那為什麼 children 不會 re-render 呢？

他不是也是子組件嗎？

這是一個蠻有趣的問題啊。原因其實很簡單，

因為 DetailsToggle、AnalyticsComponent、ChartComponent 是在父組件中宣告的，它並不會受到子組件的影響：

換句話說，子組件狀態改變，造成 Re-render 時，是不會影響到父組件的，所以我們可以將比較複雜的組件，移到上層並用 children 傳進來，這樣就可以避免 re-render 的問題。

這就是內容上移核心

如果之後有比較複雜的 layout 組件，像是 `header` 和 `floatingBar`，而且不能用狀態下移時，就可以思考看看能不能將複雜的組件移到上層，並利用 `children` 傳進來。
-->
